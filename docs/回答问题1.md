# core/task_dispatcher.py

import os

TASK_MODE = os.getenv("TASK_MODE", "local")  # local / queue

def dispatch_task(func, *args, **kwargs):
    if TASK_MODE == "local":
        return run_local(func, *args, **kwargs)
    else:
        return run_queue(func, *args, **kwargs)
4.2 本地模式（你现在的方案）
import asyncio

def run_local(func, *args, **kwargs):
    loop = asyncio.get_event_loop()
    task = loop.run_in_executor(None, func, *args)
    return {"task_id": "local_" + str(id(task))}
4.3 队列模式（RQ）
from rq import Queue
from redis import Redis

redis_conn = Redis()
queue = Queue(connection=redis_conn)

def run_queue(func, *args, **kwargs):
    job = queue.enqueue(func, *args, **kwargs)
    return {"task_id": job.id}
